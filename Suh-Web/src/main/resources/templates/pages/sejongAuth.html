        API 사용법
      

      

        

          

            
// 1. 의존성 추가 (Gradle)

            
implementation 'kr.suhsaechan:sejong-univ-auth:1.2.0'

            

            
// 2. 통합 인증 사용

            
@Autowired

            
private SuhSejongAuthEngine authEngine;

            

            
SejongAuthResult result = authEngine.authenticate("학번", "비밀번호");

            
String name = result.getName();       // "홍길동"

            
String major = result.getMajor();     // "컴퓨터공학과"
